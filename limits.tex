\section{Definitions}

\begin{definition}[Diagram of shape]
\label{def:diagram_of_shape}
Let $\cat{A}$ and $\cat{B}$ are 2 categories. The \textit{diagram of
  shape} $\cat{A}$ in $\cat{B}$ if a \mynameref{def:functor}
\[
F: \cat{A} \tof \cat{B}
\]
\end{definition}

\begin{definition}[Cone]
\label{def:cone}
Let $F$ is a \mynameref{def:diagram_of_shape} $\cat{A}$ in $\cat{B}$.
A \textit{cone} to $F$ is an object $b \in \catob{B}$ with
\mynameref{def:morphism}s $f_a: b \to F(a)$ indexed by objects from
$\cat{A}$. The cone is denoted as $cone(b, f)$.
\end{definition}
